309 template <class InnerNode>
310 class Compare {
311 public:
312  using PointDescription =
313  std::pair<const typename InnerNode::KeyT, const int_fast8_t>;
314 
315  bool
316  operator()(const InnerNode & lhs, const InnerNode & rhs) const
317  {
318  // std::cout << "Comparing points: " << lhs.get_point() << " vs " <<
319  // rhs.get_point() << "\n";
320 #pragma GCC diagnostic push
321 #pragma GCC diagnostic ignored "-Wfloat-equal"
322  if (lhs.get_point() != rhs.get_point()) {
323 #pragma GCC diagnostic pop
324  return lhs.get_point() < rhs.get_point();
325  } else {
326  /*
327  * At the same point, the order is: open ends, closed starts, closed ends,
328  * open starts
329  */
330  int_fast8_t lhs_priority;
331  if (lhs.is_closed()) {
332  if (lhs.is_start()) {
333  lhs_priority = -1;
334  } else {
335  lhs_priority = 1;
336  }
337  } else {
338  if (lhs.is_start()) {
339  lhs_priority = 2;
340  } else {
341  lhs_priority = -2;
342  }
343  }
344 
345  int_fast8_t rhs_priority;
346  if (rhs.is_closed()) {
347  if (rhs.is_start()) {
348  rhs_priority = -1;
349  } else {
350  rhs_priority = 1;
351  }
352  } else {
353  if (rhs.is_start()) {
354  rhs_priority = 2;
355  } else {
356  rhs_priority = -2;
357  }
358  }
359 
360  return lhs_priority < rhs_priority;
361  }
362  }
363 
364  bool
365  operator()(const typename InnerNode::KeyT & lhs, const InnerNode & rhs) const
366  {
367  // std::cout << "A Comparing points: " << lhs << " vs " <<
368  // rhs.get_point() << "\n";
369 #pragma GCC diagnostic push
370 #pragma GCC diagnostic ignored "-Wfloat-equal"
371  if (lhs != rhs.get_point()) {
372  return lhs < rhs.get_point();
373  }
374 #pragma GCC diagnostic pop
375 
376  // only open starts are strictly larger than the key
377  return (rhs.is_start() && !rhs.is_closed());
378  }
379 
380  bool
381  operator()(const InnerNode & lhs, const typename InnerNode::KeyT & rhs) const
382  {
383  // std::cout << "B Comparing points: " << lhs.get_point() << " vs " <<
384  // rhs << "\n";
385 #pragma GCC diagnostic push
386 #pragma GCC diagnostic ignored "-Wfloat-equal"
387  if (lhs.get_point() != rhs) {
388  return lhs.get_point() < rhs;
389  }
390 #pragma GCC diagnostic pop
391 
392  // only open ends must strictly go before the key
393  return (!lhs.is_start() && !lhs.is_closed());
394  }
395 
396  bool
397  operator()(const PointDescription & lhs, const InnerNode & rhs) const
398  {
399 #pragma GCC diagnostic push
400 #pragma GCC diagnostic ignored "-Wfloat-equal"
401  if (lhs.first != rhs.get_point()) {
402  return lhs.first < rhs.get_point();
403  }
404 #pragma GCC diagnostic pop
405 
406  if (lhs.second > 0) {
407  // the query is left-open, i.e., it should not to before anything
408  return false;
409  } else if (lhs.second < 0) {
410  // the query is right-open. It should go before everything but open end
411  // nodes
412  return rhs.is_start() || rhs.is_closed();
413  } else {
414  // The query is closed.
415  // only open starts are strictly larger than the key
416  return (rhs.is_start() && !rhs.is_closed());
417  }
418  }
419 
420  bool
421  operator()(const InnerNode & lhs, const PointDescription & rhs) const
422  {
423 #pragma GCC diagnostic push
424 #pragma GCC diagnostic ignored "-Wfloat-equal"
425  if (lhs.get_point() != rhs.first) {
426  return lhs.get_point() < rhs.first;
427  }
428 #pragma GCC diagnostic pop
429 
430  if (rhs.second > 0) {
431  // the query is left-open, i.e., everything but left-open is before it
432  return !(lhs.is_start() && !lhs.is_closed());
433  } else if (rhs.second > 0) {
434  // the query is right-open, i.e., nothing must ever strictly go before it
435  return false;
436  } else {
437  // the query is closed
438  // only open ends must strictly go before the key
439  return (!lhs.is_start() && !lhs.is_closed());
440  }
441  }
442 };
