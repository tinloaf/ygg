248 template <class InnerNode>
249 class Compare {
250 public:
251  using PointDescription =
252  std::pair<const typename InnerNode::KeyT, const int_fast8_t>;
253 
254  bool
255  operator()(const InnerNode & lhs, const InnerNode & rhs) const
256  {
257  // std::cout << "Comparing points: " << lhs.get_point() << " vs " <<
258  // rhs.get_point() << "\n";
259 #pragma GCC diagnostic push
260 #pragma GCC diagnostic ignored "-Wfloat-equal"
261  if (lhs.get_point() != rhs.get_point()) {
262 #pragma GCC diagnostic pop
263  return lhs.get_point() < rhs.get_point();
264  } else {
265  /*
266  * At the same point, the order is: open ends, closed starts, closed ends,
267  * open starts
268  */
269  int_fast8_t lhs_priority;
270  if (lhs.is_closed()) {
271  if (lhs.is_start()) {
272  lhs_priority = -1;
273  } else {
274  lhs_priority = 1;
275  }
276  } else {
277  if (lhs.is_start()) {
278  lhs_priority = 2;
279  } else {
280  lhs_priority = -2;
281  }
282  }
283 
284  int_fast8_t rhs_priority;
285  if (rhs.is_closed()) {
286  if (rhs.is_start()) {
287  rhs_priority = -1;
288  } else {
289  rhs_priority = 1;
290  }
291  } else {
292  if (rhs.is_start()) {
293  rhs_priority = 2;
294  } else {
295  rhs_priority = -2;
296  }
297  }
298 
299  return lhs_priority < rhs_priority;
300  }
301  }
302 
303  bool
304  operator()(const typename InnerNode::KeyT & lhs, const InnerNode & rhs) const
305  {
306  // std::cout << "A Comparing points: " << lhs << " vs " <<
307  // rhs.get_point() << "\n";
308 #pragma GCC diagnostic push
309 #pragma GCC diagnostic ignored "-Wfloat-equal"
310  if (lhs != rhs.get_point()) {
311  return lhs < rhs.get_point();
312  }
313 #pragma GCC diagnostic pop
314 
315  // only open starts are strictly larger than the key
316  return (rhs.is_start() && !rhs.is_closed());
317  }
318 
319  bool
320  operator()(const InnerNode & lhs, const typename InnerNode::KeyT & rhs) const
321  {
322  // std::cout << "B Comparing points: " << lhs.get_point() << " vs " <<
323  // rhs << "\n";
324 #pragma GCC diagnostic push
325 #pragma GCC diagnostic ignored "-Wfloat-equal"
326  if (lhs.get_point() != rhs) {
327  return lhs.get_point() < rhs;
328  }
329 #pragma GCC diagnostic pop
330 
331  // only open ends must strictly go before the key
332  return (!lhs.is_start() && !lhs.is_closed());
333  }
334 
335  bool
336  operator()(const PointDescription & lhs, const InnerNode & rhs) const
337  {
338 #pragma GCC diagnostic push
339 #pragma GCC diagnostic ignored "-Wfloat-equal"
340  if (lhs.first != rhs.get_point()) {
341  return lhs.first < rhs.get_point();
342  }
343 #pragma GCC diagnostic pop
344 
345  if (lhs.second > 0) {
346  // the query is left-open, i.e., it should not to before anything
347  return false;
348  } else if (lhs.second < 0) {
349  // the query is right-open. It should go before everything but open end
350  // nodes
351  return rhs.is_start() || rhs.is_closed();
352  } else {
353  // The query is closed.
354  // only open starts are strictly larger than the key
355  return (rhs.is_start() && !rhs.is_closed());
356  }
357  }
358 
359  bool
360  operator()(const InnerNode & lhs, const PointDescription & rhs) const
361  {
362 #pragma GCC diagnostic push
363 #pragma GCC diagnostic ignored "-Wfloat-equal"
364  if (lhs.get_point() != rhs.first) {
365  return lhs.get_point() < rhs.first;
366  }
367 #pragma GCC diagnostic pop
368 
369  if (rhs.second > 0) {
370  // the query is left-open, i.e., everything but left-open is before it
371  return !(lhs.is_start() && !lhs.is_closed());
372  } else if (rhs.second > 0) {
373  // the query is right-open, i.e., nothing must ever strictly go before it
374  return false;
375  } else {
376  // the query is closed
377  // only open ends must strictly go before the key
378  return (!lhs.is_start() && !lhs.is_closed());
379  }
380  }
381 };
